# Requirements
> * iOS 8.0+
> * Swift 4.0+
# Usage
```swift
func application(_ application: UIApplication, didFinishLaunchingWithOptions launchOptions: [UIApplicationLaunchOptionsKey: Any]?) -> Bool {
    self.window?.addGestureRecognizer(TouchPointGesture())
    return true
}

```
